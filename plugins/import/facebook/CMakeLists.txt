SET(QT_USE_QTWEBKIT true)
INCLUDE_DIRECTORIES(${TulipPythonInclude})
QT4_WRAP_CPP(FacebookImport_MOC_SRCS FacebookImport.h)
QT4_WRAP_UI(FacebookImport_UI_SRCS FacebookImport.ui)
QT4_ADD_RESOURCES(FacebookImport_RCC_SRCS FacebookImport.qrc)
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})
ADD_LIBRARY(FacebookImport-${TulipVersion} SHARED FacebookImport.cpp ${FacebookImport_MOC_SRCS} ${FacebookImport_RCC_SRCS} ${FacebookImport_UI_SRCS})
TARGET_LINK_LIBRARIES(FacebookImport-${TulipVersion} ${LibTulipCoreName} ${LibTulipPythonName} ${QT_LIBRARIES})
TARGET_LINK_LIBRARIES(FacebookImport-${TulipVersion} ${QT_QTWEBKIT_LIBRARY})
IF(LINUX)
TARGET_LINK_LIBRARIES(FacebookImport-${TulipVersion} ${QT_QTWEBKIT_LIBRARY_RELEASE})
ENDIF(LINUX)

IF(WIN32)
INSTALL(FILES ${CMAKE_CURRENT_SOURCE_DIR}/facebook.py DESTINATION ${TulipBinInstallDir}/python COMPONENT FacebookImport)
INSTALL(FILES ${CMAKE_CURRENT_SOURCE_DIR}/tulip_facebook.py DESTINATION ${TulipBinInstallDir}/python COMPONENT FacebookImport)
ELSE(WIN32)
INSTALL(FILES ${CMAKE_CURRENT_SOURCE_DIR}/facebook.py DESTINATION ${TulipLibInstallDir}/python COMPONENT FacebookImport)
INSTALL(FILES ${CMAKE_CURRENT_SOURCE_DIR}/tulip_facebook.py DESTINATION ${TulipLibInstallDir}/python COMPONENT FacebookImport)
ENDIF()

INSTALL(TARGETS FacebookImport-${TulipVersion} DESTINATION ${TulipPluginsInstallDir} COMPONENT FacebookImport)
